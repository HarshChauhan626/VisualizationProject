Problem Statement

Schedule tasks with a cooldown period between the same tasks using a heap-based approach.

Solution Approach

  • Heap

Pseudo Code

function leastInterval(tasks, n) {
    // Create frequency map
    const freqMap = {};
    for (const task of tasks) {
        freqMap[task] = (freqMap[task] || 0) + 1;
    }
    
    // Create max heap
    const maxHeap = new MaxHeap();
    for (const count of Object.values(freqMap)) {
        maxHeap.insert(count);
    }
    
    let time = 0;
    const waitQueue = [];
    
    while (!maxHeap.isEmpty() || !waitQueue.length === 0) {
        if (!maxHeap.isEmpty()) {
            // Get most frequent task
            let count = maxHeap.extractMax();
            
            if (count > 1) {
                // Schedule next occurrence
                waitQueue.push([count - 1, time + n]);
            }
        }
        
        time++;
        
        // Check if any task is ready to be pushed back to heap
        if (waitQueue.length > 0 && waitQueue[0][1] === time) {
            const [count, _] = waitQueue.shift();
            maxHeap.insert(count);
        }
    }
    
    return time;
}
