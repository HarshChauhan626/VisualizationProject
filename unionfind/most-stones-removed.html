Problem Statement

Remove the maximum number of stones from a grid such that we can't remove all stones in a row or column using Union-Find.

Solution Approach

  • Union-Find

Pseudo Code

function removeStones(stones) {
    // Create Union-Find data structure
    const parent = {};
    
    // Find function with path compression
    function find(x) {
        if (parent[x] !== x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    
    // Union function
    function union(x, y) {
        const rootX = find(x);
        const rootY = find(y);
        if (rootX !== rootY) {
            parent[rootY] = rootX;
        }
    }
    
    // Union stones by row and column
    for (const [x, y] of stones) {
        // Offset column index to avoid row/column conflicts
        const offset = 10000;
        
        // Initialize parent if not already
        if (!(x in parent)) parent[x] = x;
        if (!(y + offset in parent)) parent[y + offset] = y + offset;
        
        // Union row and column
        union(x, y + offset);
    }
    
    // Count unique components
    const components = new Set();
    for (const [x, y] of stones) {
        // Offset column index to avoid row/column conflicts
        const offset = 10000;
        components.add(find(x));
        components.add(find(y + offset));
    }
    
    // The maximum number of stones that can be removed
    // is total stones minus the number of unique components
    return stones.length - (components.size / 2);
}
