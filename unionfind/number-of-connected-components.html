Problem Statement

Find the number of connected components in an undirected graph using Union-Find.

Solution Approach

  • Union-Find

Pseudo Code

function countComponents(n, edges) {
    // Initialize parent array
    const parent = Array(n).fill(0).map((_, i) => i);
    
    // Find function with path compression
    function find(x) {
        if (parent[x] !== x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    
    // Union function
    function union(x, y) {
        const rootX = find(x);
        const rootY = find(y);
        if (rootX !== rootY) {
            parent[rootY] = rootX;
            return true;
        }
        return false;
    }
    
    let components = n;
    
    // Process each edge
    for (const [u, v] of edges) {
        if (union(u, v)) {
            components--;
        }
    }
    
    return components;
}
