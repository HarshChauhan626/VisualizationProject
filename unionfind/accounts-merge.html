Problem Statement

Merge accounts with common email addresses using Union-Find.

Pseudo Code

function accountsMerge(accounts) {
    // Create Union-Find data structure
    const parent = {};
    
    // Find function with path compression
    function find(email) {
        if (parent[email] !== email) {
            parent[email] = find(parent[email]);
        }
        return parent[email];
    }
    
    // Union function
    function union(email1, email2) {
        const root1 = find(email1);
        const root2 = find(email2);
        if (root1 !== root2) {
            parent[root2] = root1;
        }
    }
    
    // Initialize parent map and process accounts
    const emailToName = {};
    
    // First pass: set up parent pointers and email to name mapping
    for (const account of accounts) {
        const name = account[0];
        const mainEmail = account[1];
        
        // If this is the first time seeing any email, set it as its own parent
        if (!parent[mainEmail]) {
            parent[mainEmail] = mainEmail;
        }
        
        // Map email to name
        emailToName[mainEmail] = name;
        
        // Process additional emails
        for (let i = 2; i < account.length; i++) {
            const email = account[i];
            
            // If this is the first time seeing any email, set it as its own parent
            if (!parent[email]) {
                parent[email] = email;
            }
            
            // Union current email with main email
            union(mainEmail, email);
            
            // Map email to name
            emailToName[email] = name;
        }
    }
    
    // Group emails by their root parent
    const rootToEmails = {};
    for (const email of Object.keys(parent)) {
        const root = find(email);
        if (!rootToEmails[root]) {
            rootToEmails[root] = [];
        }
        rootToEmails[root].push(email);
    }
    
    // Format the output
    const result = [];
    for (const root of Object.keys(rootToEmails)) {
        const name = emailToName[root];
        const emails = rootToEmails[root].sort();
        result.push([name, ...emails]);
    }
    
    return result;
}
