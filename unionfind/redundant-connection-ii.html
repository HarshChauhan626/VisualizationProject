Problem Statement

Find the redundant connection in a directed graph using Union-Find.

Solution Approach

  • Union-Find

Pseudo Code

function findRedundantDirectedConnection(edges) {
    // Initialize parent array
    const parent = {};
    let conflict = [-1, -1];
    let cycle = [-1, -1];
    let candidate1 = -1;
    let candidate2 = -1;
    
    // First pass to detect conflicts
    for (const [u, v] of edges) {
        if (v in parent) {
            // Already has a parent
            conflict = [u, v];
            candidate1 = parent[v];
            candidate2 = u;
            continue;
        }
        parent[v] = u;
    }
    
    // Find function with path compression
    function find(x) {
        if (parent[x] !== x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];
    }
    
    // Second pass to detect cycles
    for (const [u, v] of edges) {
        if (candidate2 !== -1 && u === candidate2 && v === conflict[1]) {
            // Skip the second parent for the conflicting edge
            continue;
        }
        
        // If a cycle is found
        if (find(u) === v) {
            // If no conflict, it's a simple cycle
            if (conflict[0] === -1) {
                cycle = [u, v];
            } else {
                // If conflict exists, return the first parent
                return [candidate1, conflict[1]];
            }
        }
        
        // Union the nodes
        parent[v] = parent[v] === u ? parent[v] : parent[v];
    }
    
    // If a conflict was found but no cycle, return the second parent
    if (conflict[0] !== -1) {
        return [candidate2, conflict[1]];
    }
    
    // Otherwise return the edge that creates a cycle
    return cycle;
}
