Meeting Rooms II Visualization

Problem Statement

Find the minimum number of meeting rooms required using a heap-based approach.

Solution Approach

  • Heap
  • Chronological Ordering

Pseudo Code

function minMeetingRooms(intervals) {
    // Sort intervals by start time
    intervals.sort((a, b) => a[0] - b[0]);
    
    // Create min heap to track end times
    const minHeap = new MinHeap();
    
    // Add first interval's end time
    minHeap.insert(intervals[0][1]);
    
    // Process remaining intervals
    for (let i = 1; i < intervals.length; i++) {
        // If current interval's start time is >= earliest end time
        if (intervals[i][0] >= minHeap.peek()) {
            // Reuse room, update end time
            minHeap.extractMin();
        }
        
        // Add current interval's end time
        minHeap.insert(intervals[i][1]);
    }
    
    // Heap size is the minimum number of rooms needed
    return minHeap.size();
}
