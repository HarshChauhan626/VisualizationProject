Problem Statement

Find the k closest points to the origin using a max-heap approach.

Solution Approach

  • Heap

Pseudo Code

function kClosest(points, k) {
    // Create max heap
    const maxHeap = new MaxHeap();
    
    // Add first k points to the heap
    for (let i = 0; i < k; i++) {
        const [x, y] = points[i];
        const distance = x * x + y * y;
        maxHeap.insert([distance, x, y]);
    }
    
    // Process remaining points
    for (let i = k; i < points.length; i++) {
        const [x, y] = points[i];
        const distance = x * x + y * y;
        
        // If current distance is smaller than max in heap
        if (distance < maxHeap.peek()[0]) {
            maxHeap.extractMax();
            maxHeap.insert([distance, x, y]);
        }
    }
    
    // Return k closest points
    const result = [];
    while (!maxHeap.isEmpty()) {
        const [_, x, y] = maxHeap.extractMax();
        result.push([x, y]);
    }
    
    return result;
}
