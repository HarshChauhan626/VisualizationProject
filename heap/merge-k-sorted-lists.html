Merge K Sorted Lists Visualization

Problem Statement

Merge k sorted linked lists using a min-heap approach.

Solution Approach

  • Heap
  • Divide & Conquer
  • Brute Force

Pseudo Code

function mergeKLists(lists) {
    // Create a min-heap
    const minHeap = new MinHeap();
    
    // Add first element from each list
    for (let i = 0; i < lists.length; i++) {
        if (lists[i]) {
            minHeap.insert([lists[i].val, i]);
        }
    }
    
    const dummy = new ListNode(0);
    let tail = dummy;
    
    // Process elements in heap
    while (!minHeap.isEmpty()) {
        const [val, i] = minHeap.extractMin();
        tail.next = new ListNode(val);
        tail = tail.next;
        
        if (lists[i] && lists[i].next) {
            lists[i] = lists[i].next;
            minHeap.insert([lists[i].val, i]);
        }
    }
    
    return dummy.next;
}
