Ugly Number II Visualization

Problem Statement

Find the nth ugly number using a heap-based approach.

Solution Approach

  • Heap
  • Multiple Pointers

Pseudo Code

function nthUglyNumber(n, primes) {
    // Create min heap
    const minHeap = new MinHeap();
    const seen = new Set();
    
    // Initialize heap with first primes
    for (const prime of primes) {
        minHeap.insert(prime);
        seen.add(prime);
    }
    
    let current = 1;
    
    // Generate ugly numbers
    for (let i = 1; i < n; i++) {
        current = minHeap.extractMin();
        
        for (const prime of primes) {
            const next = current * prime;
            if (!seen.has(next)) {
                minHeap.insert(next);
                seen.add(next);
            }
        }
    }
    
    return current;
}
